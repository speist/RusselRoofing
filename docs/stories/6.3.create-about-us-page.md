```typescript
interface TeamMember {
  id: string;
  name: string;

  certifications: string[];
  image: string;
  email?: string;
  phone?: string;

  year: number;
  title: string;

  id: string;
  name: string;
  issuer: string;
  year: number;

  foundedYear: number;

  coreValues: string[];

  awards: Award[];
  communityInvolvement: CommunityActivity[];
}
```
